 24 /**

 28  * @property {String} subjectPublicKeyRSA_hN hexadecimal string for modulus of RSA public key (DEPRECATED from jsrsasign 7.1.4)
 29  * @property {String} subjectPublicKeyRSA_hE hexadecimal string for public exponent of RSA public key (DEPRECATED from jsrsasign 7.1.4)
 30  * @property {String} hex hexacedimal string for X.509 certificate.
 31  * @property {Number} version format version (1: X509v1, 3: X509v3, otherwise: unknown) since jsrsasign 7.1.4
 32  * @author Kenji Urushima
 33  * @version 1.0.1 (08 May 2012)
 34  * @see <a href="http://kjur.github.com/jsrsasigns/">'jsrsasign'(RSA Sign JavaScript Library) home page http://kjur.github.com/jsrsasign/</a>
 35  * @description
 36  * X509 class provides following functionality:
 37  * <ul>
 38  * <li>parse X.509 certificate ASN.1 structure</li>
 39  * <li>get basic fields, extensions, signature algorithms and signature values</li>
 40  * <li>read PEM certificate</li>
 41  * </ul>
 42  *
 43  * <ul>
 44  * <li><b>TO GET FIELDS</b>
 45  *   <ul>
 46  *   <li>serial - {@link X509#getSerialNumberHex}</li>
 47  *   <li>serial - (DEPRECATED) {@link X509.getSerialNumberHex}</li>
 48  *   <li>signature algorithm field - {@link X509#getSignatureAlgorithmField}</li>
 49  *   <li>issuer - {@link X509#getIssuerHex}</li>
 50  *   <li>issuer - {@link X509#getIssuerString}</li>
 51  *   <li>notBefore - {@link X509#getNotBefore}</li>
 52  *   <li>notAfter - {@link X509#getNotAfter}</li>
 53  *   <li>subject - {@link X509#getSubjectHex}</li>
 54  *   <li>subject - {@link X509#getSubjectString}</li>
 55  *   <li>subjectPublicKeyInfo - {@link X509#.getPublicKey}</li>
 56  *   <li>subjectPublicKeyInfo - {@link X509#.getPublicKeyHex}</li>
 57  *   <li>subjectPublicKeyInfo - {@link X509#.getPublicKeyIdx}</li>

105     /**
106      * get format version (X.509v1 or v3 certificate)<br/>
107      * @name getVersion
108      * @memberOf X509#
109      * @function
110      * @return {Number} 1 for X509v1, 3 for X509v3, otherwise 0
111      * @since jsrsasign 7.1.14 x509 1.1.13
112      * @description
113      * This method returns a format version of X.509 certificate.
114      * It returns 1 for X.509v1 certificate and 3 for v3 certificate.
115      * Otherwise returns 0.
116      * This method will be automatically called in
117      * {@link X509#readCertPEM}. After then, you can use
118      * {@link X509.version} parameter.
119      * @example
120      * var x = new X509();
121      * x.readCertPEM(sCertPEM);
122      * version = x.getVersion();    // 1 or 3
123      * sn = x.getSerialNumberHex(); // return string like "01ad..."
124      */
125     this.getVersion = function() {
126 	if (this.hex === null || this.version !== 0) return this.version;
127 
128 	// check if the first item of tbsCertificate "[0] { INTEGER 2 }"
129 	if (_ASN1HEX.getTLVbyList(this.hex, 0, [0, 0]) !==
130 	    "a003020102") {
131 	    this.version = 1;
132 	    this.foffset = -1;
133 	    return 1;
134 	}
135 
136 	this.version = 3;
137 	return 3;
138     };

140     /**
141      * get hexadecimal string of serialNumber field of certificate.<br/>
142      * @name getSerialNumberHex
143      * @memberOf X509#
144      * @function
145      * @return {String} hexadecimal string of certificate serial number
146      * @example
147      * var x = new X509();
148      * x.readCertPEM(sCertPEM);
149      * var sn = x.getSerialNumberHex(); // return string like "01ad..."
150      */
151     this.getSerialNumberHex = function() {
152 	return _ASN1HEX.getVbyList(this.hex, 0, [0, 1 + this.foffset], "02");
153     };

155     /**
156      * get signature algorithm name in basic field
157      * @name getSignatureAlgorithmField
158      * @memberOf X509#
159      * @function
160      * @return {String} signature algorithm name (ex. SHA1withRSA, SHA256withECDSA)
161      * @since x509 1.1.8
162      * @description
163      * This method will get a name of signature algorithm field of certificate:
164      * @example
165      * var x = new X509();
166      * x.readCertPEM(sCertPEM);
167      * algName = x.getSignatureAlgorithmField();
168      */
169     this.getSignatureAlgorithmField = function() {
170 	var sigAlgOidHex = _ASN1HEX.getVbyList(this.hex, 0, [0, 2 + this.foffset, 0], "06");
171 	var sigAlgOidInt = KJUR.asn1.ASN1Util.oidHexToInt(sigAlgOidHex);
172 	var sigAlgName = KJUR.asn1.x509.OID.oid2name(sigAlgOidInt);
173 	return sigAlgName;
174     };

236     /**
237      * get notBefore field string of certificate.<br/>
238      * @name getNotBefore
239      * @memberOf X509#
240      * @function
241      * @return {String} not before time value (ex. "151231235959Z")
242      * @example
243      * var x = new X509();
244      * x.readCertPEM(sCertPEM);
245      * var notBefore = x.getNotBefore(); // return string like "151231235959Z"

248         var s = _ASN1HEX.getVbyList(this.hex, 0, [0, 4 + this.foffset, 0]);

269         return s;

321     /**

335     /**
336      * read a hexadecimal string of X.509 certificate<br/>
337      * @name readCertHex
338      * @memberOf X509#
339      * @function
340      * @param {String} sCertHex hexadecimal string of X.509 certificate
341      * @since jsrsasign 7.1.4 x509 1.1.13
342      * @example
343      * x = new X509();
344      * x.readCertHex("3082..."); // read certificate
345      */
346     this.readCertHex = function(sCertHex) {
347         this.hex = sCertHex;
348 	this.getVersion(); // set version parameter
349 	pubkey = this.getPublicKey();
350 
351 	// deprecated field settings. will remove this block after 8.0.0
352 	if (pubkey instanceof RSAKey) { 
353             this.subjectPublicKeyRSA = pubkey;
354             this.subjectPublicKeyRSA_hN = hextoposhex(pubkey.n.toString(16));
355             this.subjectPublicKeyRSA_hE = hextoposhex(pubkey.e.toString(16));
356 	}
357     };

363         if (typeof this.subjectPublicKeyRSA.setPublic === "function") {

366         this.subjectPublicKeyRSA_hN = a[0];
367         this.subjectPublicKeyRSA_hE = a[1];

371     /**
372      * get certificate information as string.<br/>
373      * @name getInfo
374      * @memberOf X509#
375      * @function
376      * @return {String} certificate information string
377      * @since jsrsasign 5.0.10 x509 1.1.8
378      * @example
379      * x = new X509();
380      * x.readCertPEM(certPEM);
381      * console.log(x.getInfo());
382      * // this shows as following
383      * Basic Fields
384      *   serial number: 02ac5c266a0b409b8f0b79f2ae462577
385      *   signature algorithm: SHA1withRSA
386      *   issuer: /C=US/O=DigiCert Inc/OU=www.digicert.com/CN=DigiCert High Assurance EV Root CA
387      *   notBefore: 061110000000Z
388      *   notAfter: 311110000000Z
389      *   subject: /C=US/O=DigiCert Inc/OU=www.digicert.com/CN=DigiCert High Assurance EV Root CA
390      *   subject public key info:
391      *     key algorithm: RSA
392      *     n=c6cce573e6fbd4bb...
393      *     e=10001
394      * X509v3 Extensions:
395      *   keyUsage CRITICAL:
396      *     digitalSignature,keyCertSign,cRLSign
397      *   basicConstraints CRITICAL:
398      *     cA=true
399      *   subjectKeyIdentifier :
400      *     b13ec36903f8bf4701d498261a0802ef63642bc3
401      *   authorityKeyIdentifier :
402      *     kid=b13ec36903f8bf4701d498261a0802ef63642bc3
403      * signature algorithm: SHA1withRSA
404      * signature: 1c1a0697dcd79c9f...
405      */
406     this.getInfo = function() {
407 	var _X509 = X509;
408 	var s, pubkey, aExt;
409 	s  = "Basic Fields\n";
410         s += "  serial number: " + this.getSerialNumberHex() + "\n";
411 	s += "  signature algorithm: " + this.getSignatureAlgorithmField() + "\n";
412 	s += "  issuer: " + this.getIssuerString() + "\n";
413 	s += "  notBefore: " + this.getNotBefore() + "\n";
414 	s += "  notAfter: " + this.getNotAfter() + "\n";
415 	s += "  subject: " + this.getSubjectString() + "\n";
416 	s += "  subject public key info: " + "\n";
417 
418 	// subject public key info
419 	pubkey = this.getPublicKey();
420 	s += "    key algorithm: " + pubkey.type + "\n";
421 
422 	if (pubkey.type === "RSA") {
423 	    s += "    n=" + hextoposhex(pubkey.n.toString(16)).substr(0, 16) + "...\n";
424 	    s += "    e=" + hextoposhex(pubkey.e.toString(16)) + "\n";
425 	}
426 
427         s += "X509v3 Extensions:\n";
428 
429 	aExt = _X509.getV3ExtInfoListOfCertHex(this.hex);
430         for (var i = 0; i < aExt.length; i++) {
431 	    var info = aExt[i];
432 
433 	    // show extension name and critical flag
434 	    var extName = KJUR.asn1.x509.OID.oid2name(info["oid"]);
435 	    if (extName === '') extName = info["oid"];
436 
437 	    var critical = '';
438 	    if (info["critical"] === true) critical = "CRITICAL";
439 
440 	    s += "  " + extName + " " + critical + ":\n";
441 
442 	    // show extension value if supported
443 	    if (extName === "basicConstraints") {
444 		var bc = _X509.getExtBasicConstraints(this.hex);
445 		if (bc.cA === undefined) {
446 		    s += "    {}\n";
447 		} else {
448 		    s += "    cA=true";
449 		    if (bc.pathLen !== undefined)
450 			s += ", pathLen=" + bc.pathLen;
451 		    s += "\n";
452 		}
453 	    } else if (extName === "keyUsage") {
454 		s += "    " + _X509.getExtKeyUsageString(this.hex) + "\n";
455 	    } else if (extName === "subjectKeyIdentifier") {
456 		s += "    " + _X509.getExtSubjectKeyIdentifier(this.hex) + "\n";
457 	    } else if (extName === "authorityKeyIdentifier") {
458 		var akid = _X509.getExtAuthorityKeyIdentifier(this.hex);
459 		if (akid.kid !== undefined)
460 		    s += "    kid=" + akid.kid + "\n";
461 	    } else if (extName === "extKeyUsage") {
462 		var eku = _X509.getExtExtKeyUsageName(this.hex);
463 		s += "    " + eku.join(", ") + "\n";
464 	    } else if (extName === "subjectAltName") {
465 		var san = _X509.getExtSubjectAltName(this.hex);
466 		s += "    " + san.join(", ") + "\n";
467 	    } else if (extName === "cRLDistributionPoints") {
468 		var cdp = _X509.getExtCRLDistributionPointsURI(this.hex);
469 		s += "    " + cdp + "\n";
470 	    } else if (extName === "authorityInfoAccess") {
471 		var aia = _X509.getExtAIAInfo(this.hex);
472 		if (aia.ocsp !== undefined)
473 		    s += "    ocsp: " + aia.ocsp.join(",") + "\n";
474 		if (aia.caissuer !== undefined)
475 		    s += "    caissuer: " + aia.caissuer.join(",") + "\n";
476 	    }
477         }
478 
479 	s += "signature algorithm: " + _X509.getSignatureAlgorithmName(this.hex) + "\n";
480 	s += "signature: " + _X509.getSignatureValueHex(this.hex).substr(0, 16) + "...\n";
481 	return s;
482     };

485 /**
486  * get Base64 string from PEM certificate string
487  * @name pemToBase64
488  * @memberOf X509
489  * @function
490  * @param {String} sCertPEM PEM formatted RSA/ECDSA/DSA X.509 certificate
491  * @return {String} Base64 string of PEM certificate
492  * @example
493  * b64 = X509.pemToBase64(certPEM);
494  */
495 X509.pemToBase64 = function(sCertPEM) {
496     var s = sCertPEM;
497     s = s.replace("-----BEGIN CERTIFICATE-----", "");
498     s = s.replace("-----END CERTIFICATE-----", "");
499     s = s.replace(/[ \n]+/g, "");
500     return s;
501 };

522  * @name getSubjectPublicKeyPosFromCertHex
523  * @memberOf X509

526  * @return {Integer} string index of key contents

531 // NOTE: Without BITSTRING encapsulation.
532 X509.getSubjectPublicKeyPosFromCertHex = function(hCert) {

1268  * @description
1269  * This method will get authority info access value
1270  * as associate array which has following properties:
1271  * <ul>
1272  * <li>ocsp - array of string for OCSP responder URL</li>
1273  * <li>caissuer - array of string for caIssuer value (i.e. CA certificates URL)</li>
1274  * </ul>

1277  * @example

1284     var result = {};
1285     result.ocsp = [];
1286     result.caissuer = [];

1288     if (pos1 == -1) return null;
1289     if (hCert.substr(pos1, 2) != "30") // extnValue SEQUENCE
1290 	throw "malformed AIA Extn Value";
1291 
1292     var posAccDescList = ASN1HEX.getPosArrayOfChildren_AtObj(hCert, pos1);
1293     for (var i = 0; i < posAccDescList.length; i++) {
1294 	var p = posAccDescList[i];
1295 	var posAccDescChild = ASN1HEX.getPosArrayOfChildren_AtObj(hCert, p);
1296 	if (posAccDescChild.length != 2)
1297 	    throw "malformed AccessDescription of AIA Extn";
1298 	var pOID = posAccDescChild[0];
1299 	var pName = posAccDescChild[1];
1300 	if (ASN1HEX.getHexOfV_AtObj(hCert, pOID) == "2b06010505073001") {
1301 	    if (hCert.substr(pName, 2) == "86") {
1302 		result.ocsp.push(hextoutf8(ASN1HEX.getHexOfV_AtObj(hCert, pName)));
1303 	    }
1304 	}
1305 	if (ASN1HEX.getHexOfV_AtObj(hCert, pOID) == "2b06010505073002") {
1306 	    if (hCert.substr(pName, 2) == "86") {
1307 		result.caissuer.push(hextoutf8(ASN1HEX.getHexOfV_AtObj(hCert, pName)));
1308 	    }
1309 	}
