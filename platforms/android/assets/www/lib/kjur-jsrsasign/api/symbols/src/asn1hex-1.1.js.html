 23 /*
 24  * MEMO:
 25  *   f('3082025b02...', 2) ... 82025b ... 3bytes
 26  *   f('020100', 2) ... 01 ... 1byte
 27  *   f('0203001...', 2) ... 03 ... 1byte
 28  *   f('02818003...', 2) ... 8180 ... 2bytes
 29  *   f('3080....0000', 2) ... 80 ... -1
 30  *
 31  *   Requirements:
 32  *   - ASN.1 type octet length MUST be 1. 
 33  *     (i.e. ASN.1 primitives like SET, SEQUENCE, INTEGER, OCTETSTRING ...)
 34  */

 36 /**
 37  * ASN.1 DER encoded hexadecimal string utility class
 38  * @name ASN1HEX
 39  * @class ASN.1 DER encoded hexadecimal string utility class
 40  * @since jsrsasign 1.1
 41  * @description
 42  * This class provides a parser for hexadecimal string of
 43  * DER encoded ASN.1 binary data.
 44  * Here are major methods of this class.
 45  * <ul>
 46  * <li><b>ACCESS BY POSITION</b>
 47  *   <ul>
 48  *   <li>{@link ASN1HEX.getHexOfTLV_AtObj} - get ASN.1 TLV at specified position</li>
 49  *   <li>{@link ASN1HEX.getHexOfV_AtObj} - get ASN.1 V at specified position</li>
 50  *   <li>{@link ASN1HEX.getHexOfL_AtObj} - get hexadecimal ASN.1 L at specified position</li>
 51  *   <li>{@link ASN1HEX.getIntOfL_AtObj} - get integer ASN.1 L at specified position</li>
 52  *   <li>{@link ASN1HEX.getStartPosOfV_AtObj} - get ASN.1 V position from its ASN.1 TLV position</li>
 53  *   </ul>
 54  * </li>
 55  * <li><b>ACCESS FOR CHILD ITEM</b>
 56  *   <ul>
 57  *   <li>{@link ASN1HEX.getNthChildIndex_AtObj} - get nth child index at specified position</li>
 58  *   <li>{@link ASN1HEX.getPosArrayOfChildren_AtObj} - get indexes of children</li>
 59  *   <li>{@link ASN1HEX.getPosOfNextSibling_AtObj} - get position of next sibling</li>
 60  *   </ul>
 61  * </li>
 62  * <li><b>ACCESS NESTED ASN.1 STRUCTURE</b>
 63  *   <ul>
 64  *   <li>{@link ASN1HEX.getVbyList} - get ASN.1 V at specified nth list index with checking expected tag</li>
 65  *   <li>{@link ASN1HEX.getDecendantHexTLVByNthList} - get ASN.1 TLV at specified list index</li>
 66  *   <li>{@link ASN1HEX.getDecendantHexVByNthList} - get ASN.1 V at specified list index</li>
 67  *   <li>{@link ASN1HEX.getDecendantIndexByNthList} - get index at specified list index</li>
 68  *   </ul>
 69  * </li>
 70  * <li><b>UTILITIES</b>
 71  *   <ul>
 72  *   <li>{@link ASN1HEX.dump} - dump ASN.1 structure</li>
 73  *   <li>{@link ASN1HEX.isASN1HEX} - check whether ASN.1 hexadecimal string or not</li>
 74  *   <li>{@link ASN1HEX.hextooidstr} - convert hexadecimal string of OID to dotted integer list</li>
 75  *   </ul>
 76  * </li>
 77  * </ul>
 78  */
 79 var ASN1HEX = new function() {
 80 };

 82 /**
 83  * get byte length for ASN.1 L(length) bytes<br/>
 84  * @name getByteLengthOfL_AtObj
 85  * @memberOf ASN1HEX
 86  * @function
 87  * @param {String} s hexadecimal string of ASN.1 DER encoded data
 88  * @param {Number} pos string index
 89  * @return byte length for ASN.1 L(length) bytes
 90  */
 91 ASN1HEX.getByteLengthOfL_AtObj = function(s, pos) {
 92     if (s.substring(pos + 2, pos + 3) != '8') return 1;
 93     var i = parseInt(s.substring(pos + 3, pos + 4));
 94     if (i == 0) return -1;             // length octet '80' indefinite length
 95     if (0 < i && i < 10) return i + 1; // including '8?' octet;
 96     return -2;                         // malformed format
 97 };
 98 
 99 /**
100  * get hexadecimal string for ASN.1 L(length) bytes<br/>
101  * @name getHexOfL_AtObj
102  * @memberOf ASN1HEX
103  * @function
104  * @param {String} s hexadecimal string of ASN.1 DER encoded data
105  * @param {Number} pos string index
106  * @return {String} hexadecimal string for ASN.1 L(length) bytes
107  */
108 ASN1HEX.getHexOfL_AtObj = function(s, pos) {
109     var len = ASN1HEX.getByteLengthOfL_AtObj(s, pos);
110     if (len < 1) return '';
111     return s.substring(pos + 2, pos + 2 + len * 2);
112 };
113 
114 /**
115  * get integer value of ASN.1 length for ASN.1 data<br/>
116  * @name getIntOfL_AtObj
117  * @memberOf ASN1HEX
118  * @function
119  * @param {String} s hexadecimal string of ASN.1 DER encoded data
120  * @param {Number} pos string index
121  * @return ASN.1 L(length) integer value
122  */
123 /*
124  getting ASN.1 length value at the position 'idx' of
125  hexa decimal string 's'.
126  f('3082025b02...', 0) ... 82025b ... ???
127  f('020100', 0) ... 01 ... 1
128  f('0203001...', 0) ... 03 ... 3
129  f('02818003...', 0) ... 8180 ... 128
130  */
131 ASN1HEX.getIntOfL_AtObj = function(s, pos) {
132     var hLength = ASN1HEX.getHexOfL_AtObj(s, pos);
133     if (hLength == '') return -1;
134     var bi;
135     if (parseInt(hLength.substring(0, 1)) < 8) {
136         bi = new BigInteger(hLength, 16);
137     } else {
138         bi = new BigInteger(hLength.substring(2), 16);
139     }
140     return bi.intValue();
141 };

143 /**
144  * get ASN.1 value starting string position for ASN.1 object refered by index 'idx'.
145  * @name getStartPosOfV_AtObj
146  * @memberOf ASN1HEX
147  * @function
148  * @param {String} s hexadecimal string of ASN.1 DER encoded data
149  * @param {Number} pos string index
150  */
151 ASN1HEX.getStartPosOfV_AtObj = function(s, pos) {
152     var l_len = ASN1HEX.getByteLengthOfL_AtObj(s, pos);
153     if (l_len < 0) return l_len;
154     return pos + (l_len + 1) * 2;
155 };

157 /**
158  * get hexadecimal string of ASN.1 V(value)
159  * @name getHexOfV_AtObj
160  * @memberOf ASN1HEX
161  * @function
162  * @param {String} s hexadecimal string of ASN.1 DER encoded data
163  * @param {Number} pos string index
164  * @return {String} hexadecimal string of ASN.1 value.
165  */
166 ASN1HEX.getHexOfV_AtObj = function(s, pos) {
167     var pos1 = ASN1HEX.getStartPosOfV_AtObj(s, pos);
168     var len = ASN1HEX.getIntOfL_AtObj(s, pos);
169     return s.substring(pos1, pos1 + len * 2);
170 };

172 /**
173  * get hexadecimal string of ASN.1 TLV at<br/>
174  * @name getHexOfTLV_AtObj
175  * @memberOf ASN1HEX
176  * @function
177  * @param {String} s hexadecimal string of ASN.1 DER encoded data
178  * @param {Number} pos string index
179  * @return {String} hexadecimal string of ASN.1 TLV.
180  * @since asn1hex 1.1
181  */
182 ASN1HEX.getHexOfTLV_AtObj = function(s, pos) {
183     var hT = s.substr(pos, 2);
184     var hL = ASN1HEX.getHexOfL_AtObj(s, pos);
185     var hV = ASN1HEX.getHexOfV_AtObj(s, pos);
186     return hT + hL + hV;
187 };

189 // ========== sibling methods ================================
190 /**
191  * get next sibling starting index for ASN.1 object string<br/>
192  * @name getPosOfNextSibling_AtObj
193  * @memberOf ASN1HEX
194  * @function
195  * @param {String} s hexadecimal string of ASN.1 DER encoded data
196  * @param {Number} pos string index
197  * @return next sibling starting index for ASN.1 object string
198  */
199 ASN1HEX.getPosOfNextSibling_AtObj = function(s, pos) {
200     var pos1 = ASN1HEX.getStartPosOfV_AtObj(s, pos);
201     var len = ASN1HEX.getIntOfL_AtObj(s, pos);
202     return pos1 + len * 2;
203 };

205 // ========== children methods ===============================
206 /**
207  * get array of string indexes of child ASN.1 objects<br/>
208  * @name getPosArrayOfChildren_AtObj
209  * @memberOf ASN1HEX
210  * @function
211  * @param {String} h hexadecimal string of ASN.1 DER encoded data
212  * @param {Number} pos start string index of ASN.1 object
213  * @return {Array of Number} array of indexes for childen of ASN.1 objects
214  * @description
215  * This method returns array of integers for a concatination of ASN.1 objects
216  * in a ASN.1 value. As for BITSTRING, one byte of unusedbits is skipped.
217  * As for other ASN.1 simple types such as INTEGER, OCTET STRING or PRINTABLE STRING,
218  * it returns a array of a string index of its ASN.1 value.<br/>
219  * NOTE: Since asn1hex 1.1.7 of jsrsasign 6.1.2, Encapsulated BitString is supported.
220  * @example
221  * ASN1HEX.getPosArrayOfChildren_AtObj("0203012345", 0) ⇒ [4] // INTEGER 012345
222  * ASN1HEX.getPosArrayOfChildren_AtObj("1303616161", 0) ⇒ [4] // PrintableString aaa
223  * ASN1HEX.getPosArrayOfChildren_AtObj("030300ffff", 0) ⇒ [6] // BITSTRING ffff (unusedbits=00a)
224  * ASN1HEX.getPosArrayOfChildren_AtObj("3006020104020105", 0) ⇒ [4, 10] // SEQUENCE(INT4,INT5)
225  */
226 ASN1HEX.getPosArrayOfChildren_AtObj = function(h, pos) {
227     var a = new Array();
228     var p0 = ASN1HEX.getStartPosOfV_AtObj(h, pos);
229     if (h.substr(pos, 2) == "03") {
230 	a.push(p0 + 2); // BITSTRING value without unusedbits
231     } else {
232 	a.push(p0);
233     }
234 
235     var len = ASN1HEX.getIntOfL_AtObj(h, pos);
236     var p = p0;
237     var k = 0;
238     while (1) {
239         var pNext = ASN1HEX.getPosOfNextSibling_AtObj(h, p);
240         if (pNext == null || (pNext - p0  >= (len * 2))) break;
241         if (k >= 200) break;
242             
243         a.push(pNext);
244         p = pNext;
245             
246         k++;
247     }
248     
249     return a;
250 };

257  * @param {String} h hexadecimal string of ASN.1 DER encoded data
258  * @param {Number} idx start string index of ASN.1 object

282  * NOTE: Optional checkingTag has been supported since

285  * The "nthList" is a index list of structured ASN.1 object

301 /**
302  * get string index of nth child object of ASN.1 object refered by h, idx<br/>
303  * @name getIdxbyList
304  * @memberOf ASN1HEX
305  * @function
306  * @param {String} h hexadecimal string of ASN.1 DER encoded data
307  * @param {Number} currentIndex start string index of ASN.1 object
308  * @param {Array of Number} nthList array list of nth
309  * @param {String} checkingTag (OPTIONAL) string of expected ASN.1 tag for nthList 
310  * @return {Number} string index refered by nthList
311  * @since jsrsasign 7.1.4 asn1hex 1.1.10.
312  * @description
313  * @example
314  * The "nthList" is a index list of structured ASN.1 object
315  * reference. Here is a sample structure and "nthList"s which
316  * refers each objects.
317  *
318  * SQUENCE               - 
319  *   SEQUENCE            - [0]
320  *     IA5STRING 000     - [0, 0]
321  *     UTF8STRING 001    - [0, 1]
322  *   SET                 - [1]
323  *     IA5STRING 010     - [1, 0]
324  *     UTF8STRING 011    - [1, 1]
325  */
326 ASN1HEX.getIdxbyList = function(h, currentIndex, nthList, checkingTag) {
327     var _ASN1HEX = ASN1HEX;
328     var firstNth, a;
329     if (nthList.length == 0) {
330 	if (checkingTag !== undefined) {
331             if (h.substr(currentIndex, 2) !== checkingTag) {
332 		throw "checking tag doesn't match: " + 
333                     h.substr(currentIndex, 2) + "!=" + checkingTag;
334             }
335 	}
336         return currentIndex;
337     }
338     firstNth = nthList.shift();
339     a = _ASN1HEX.getPosArrayOfChildren_AtObj(h, currentIndex);
340     return _ASN1HEX.getIdxbyList(h, a[firstNth], nthList, checkingTag);
341 };

376 /**
377  * get ASN.1 value by nthList<br/>
378  * @name getVbyList
379  * @memberOf ASN1HEX
380  * @function
381  * @param {String} h hexadecimal string of ASN.1 structure
382  * @param {Integer} currentIndex string index to start searching in hexadecimal string "h"
383  * @param {Array} nthList array of nth list index
384  * @param {String} checkingTag (OPTIONAL) string of expected ASN.1 tag for nthList 
385  * @param {Boolean} removeUnusedbits (OPTIONAL) flag for remove first byte for value (DEFAULT false)
386  * @description
387  * This static method is to get a ASN.1 value which specified "nthList" position
388  * with checking expected tag "checkingTag".
389  * NOTE: 'removeUnusedbits' flag has been supported since
390  * jsrsasign 7.1.14 asn1hex 1.1.10.
391  * @since asn1hex 1.1.4
392  */
393 ASN1HEX.getVbyList = function(h, currentIndex, nthList, checkingTag, removeUnusedbits) {
394     var _ASN1HEX = ASN1HEX;
395     var idx, v;
396     idx = _ASN1HEX.getIdxbyList(h, currentIndex, nthList, checkingTag);
397     
398     if (idx === undefined) {
399         throw "can't find nthList object";
400     }
401 
402     v = _ASN1HEX.getHexOfV_AtObj(h, idx);
403     if (removeUnusedbits === true) v = v.substr(2);
404     return v;
405 };

407 /**
408  * get ASN.1 TLV by nthList<br/>
409  * @name getTLVbyList
410  * @memberOf ASN1HEX
411  * @function
412  * @param {String} h hexadecimal string of ASN.1 structure
413  * @param {Integer} currentIndex string index to start searching in hexadecimal string "h"
414  * @param {Array} nthList array of nth list index
415  * @param {String} checkingTag (OPTIONAL) string of expected ASN.1 tag for nthList 
416  * @since jsrsasign 7.1.4 asn1hex 1.1.10
417  * @description
418  * This static method is to get a ASN.1 value which specified "nthList" position
419  * with checking expected tag "checkingTag".
420  */
421 ASN1HEX.getTLVbyList = function(h, currentIndex, nthList, checkingTag) {
422     var _ASN1HEX = ASN1HEX;
423     var idx = _ASN1HEX.getDecendantIndexByNthList(h, currentIndex, nthList);
424     if (idx === undefined) {
425         throw "can't find nthList object";
426     }
427     if (checkingTag !== undefined) {
428         if (h.substr(idx, 2) != checkingTag) {
429             throw "checking tag doesn't match: " + 
430                 h.substr(idx,2) + "!=" + checkingTag;
431         }
432     }
433     return _ASN1HEX.getHexOfTLV_AtObj(h, idx);
434 };

436 /**
437  * get OID string from hexadecimal encoded value<br/>
438  * @name hextooidstr
439  * @memberOf ASN1HEX
440  * @function
441  * @param {String} hex hexadecmal string of ASN.1 DER encoded OID value
442  * @return {String} OID string (ex. '1.2.3.4.567')
443  * @since asn1hex 1.1.5
444  */
445 ASN1HEX.hextooidstr = function(hex) {
446     var zeroPadding = function(s, len) {
447         if (s.length >= len) return s;
448         return new Array(len - s.length + 1).join('0') + s;
449     };
450 
451     var a = [];
452 
453     // a[0], a[1]
454     var hex0 = hex.substr(0, 2);
455     var i0 = parseInt(hex0, 16);
456     a[0] = new String(Math.floor(i0 / 40));
457     a[1] = new String(i0 % 40);
458 
459     // a[2]..a[n]
460    var hex1 = hex.substr(2);
461     var b = [];
462     for (var i = 0; i < hex1.length / 2; i++) {
463     b.push(parseInt(hex1.substr(i * 2, 2), 16));
464     }
465     var c = [];
466     var cbin = "";
467     for (var i = 0; i < b.length; i++) {
468         if (b[i] & 0x80) {
469             cbin = cbin + zeroPadding((b[i] & 0x7f).toString(2), 7);
470         } else {
471             cbin = cbin + zeroPadding((b[i] & 0x7f).toString(2), 7);
472             c.push(new String(parseInt(cbin, 2)));
473             cbin = "";
474         }
475     }
476 
477     var s = a.join(".");
478     if (c.length > 0) s = s + "." + c.join(".");
479     return s;
480 };

482 /**
483  * get string of simple ASN.1 dump from hexadecimal ASN.1 data<br/>
484  * @name dump
485  * @memberOf ASN1HEX
486  * @function
487  * @param {Object} hexOrObj hexadecmal string of ASN.1 data or ASN1Object object
488  * @param {Array} flags associative array of flags for dump (OPTION)
489  * @param {Number} idx string index for starting dump (OPTION)
490  * @param {String} indent indent string (OPTION)
491  * @return {String} string of simple ASN.1 dump
492  * @since jsrsasign 4.8.3 asn1hex 1.1.6
493  * @description
494  * This method will get an ASN.1 dump from
495  * hexadecmal string of ASN.1 DER encoded data.
496  * Here are features:
497  * <ul>
498  * <li>ommit long hexadecimal string</li>
499  * <li>dump encapsulated OCTET STRING (good for X.509v3 extensions)</li>
500  * <li>structured/primitive context specific tag support (i.e. [0], [3] ...)</li>
501  * <li>automatic decode for implicit primitive context specific tag 
502  * (good for X.509v3 extension value)
503  *   <ul>
504  *   <li>if hex starts '68747470'(i.e. http) it is decoded as utf8 encoded string.</li>
505  *   <li>if it is in 'subjectAltName' extension value and is '[2]'(dNSName) tag
506  *   value will be encoded as utf8 string</li>
507  *   <li>otherwise it shows as hexadecimal string</li>
508  *   </ul>
509  * </li>
510  * </ul>
511  * NOTE1: Argument {@link KJUR.asn1.ASN1Object} object is supported since
512  * jsrsasign 6.2.4 asn1hex 1.0.8
513  * @example
514  * // 1) ASN.1 INTEGER
515  * ASN1HEX.dump('0203012345')
516  * ↓
517  * INTEGER 012345
518  *
519  * // 2) ASN.1 Object Identifier
520  * ASN1HEX.dump('06052b0e03021a')
521  * ↓
522  * ObjectIdentifier sha1 (1 3 14 3 2 26)
523  *
524  * // 3) ASN.1 SEQUENCE
525  * ASN1HEX.dump('3006020101020102')
526  * ↓
527  * SEQUENCE
528  *   INTEGER 01
529  *   INTEGER 02
530  *
531  * // 4) ASN.1 SEQUENCE since jsrsasign 6.2.4
532  * o = KJUR.asn1.ASN1Util.newObject({seq: [{int: 1}, {int: 2}]});
533  * ASN1HEX.dump(o)
534  * ↓
535  * SEQUENCE
536  *   INTEGER 01
537  *   INTEGER 02
538  * // 5) ASN.1 DUMP FOR X.509 CERTIFICATE
539  * ASN1HEX.dump(ASN1HEX.pemToHex(certPEM))
540  * ↓
541  * SEQUENCE
542  *   SEQUENCE
543  *     [0]
544  *       INTEGER 02
545  *     INTEGER 0c009310d206dbe337553580118ddc87
546  *     SEQUENCE
547  *       ObjectIdentifier SHA256withRSA (1 2 840 113549 1 1 11)
548  *       NULL
549  *     SEQUENCE
550  *       SET
551  *         SEQUENCE
552  *           ObjectIdentifier countryName (2 5 4 6)
553  *           PrintableString 'US'
554  *             :
555  */
556 ASN1HEX.dump = function(hexOrObj, flags, idx, indent) {
557     var hex = hexOrObj;
558     if (hexOrObj instanceof KJUR.asn1.ASN1Object)
559 	hex = hexOrObj.getEncodedHex();
560 
561     var _skipLongHex = function(hex, limitNumOctet) {
562 	if (hex.length <= limitNumOctet * 2) {
563 	    return hex;
564 	} else {
565 	    var s = hex.substr(0, limitNumOctet) + 
566 		    "..(total " + hex.length / 2 + "bytes).." +
567 		    hex.substr(hex.length - limitNumOctet, limitNumOctet);
568 	    return s;
569 	};
570     };
571 
572     if (flags === undefined) flags = { "ommit_long_octet": 32 };
573     if (idx === undefined) idx = 0;
574     if (indent === undefined) indent = "";
575     var skipLongHex = flags.ommit_long_octet;
576 
577     if (hex.substr(idx, 2) == "01") {
578 	var v = ASN1HEX.getHexOfV_AtObj(hex, idx);
579 	if (v == "00") {
580 	    return indent + "BOOLEAN FALSE\n";
581 	} else {
582 	    return indent + "BOOLEAN TRUE\n";
583 	}
584     }
585     if (hex.substr(idx, 2) == "02") {
586 	var v = ASN1HEX.getHexOfV_AtObj(hex, idx);
587 	return indent + "INTEGER " + _skipLongHex(v, skipLongHex) + "\n";
588     }
589     if (hex.substr(idx, 2) == "03") {
590 	var v = ASN1HEX.getHexOfV_AtObj(hex, idx);
591 	return indent + "BITSTRING " + _skipLongHex(v, skipLongHex) + "\n";
592     }
593     if (hex.substr(idx, 2) == "04") {
594 	var v = ASN1HEX.getHexOfV_AtObj(hex, idx);
595 	if (ASN1HEX.isASN1HEX(v)) {
596 	    var s = indent + "OCTETSTRING, encapsulates\n";
597 	    s = s + ASN1HEX.dump(v, flags, 0, indent + "  ");
598 	    return s;
599 	} else {
600 	    return indent + "OCTETSTRING " + _skipLongHex(v, skipLongHex) + "\n";
601 	}
602     }
603     if (hex.substr(idx, 2) == "05") {
604 	return indent + "NULL\n";
605     }
606     if (hex.substr(idx, 2) == "06") {
607 	var hV = ASN1HEX.getHexOfV_AtObj(hex, idx);
608         var oidDot = KJUR.asn1.ASN1Util.oidHexToInt(hV);
609         var oidName = KJUR.asn1.x509.OID.oid2name(oidDot);
610 	var oidSpc = oidDot.replace(/\./g, ' ');
611         if (oidName != '') {
612   	    return indent + "ObjectIdentifier " + oidName + " (" + oidSpc + ")\n";
613 	} else {
614   	    return indent + "ObjectIdentifier (" + oidSpc + ")\n";
615 	}
616     }
617     if (hex.substr(idx, 2) == "0c") {
618 	return indent + "UTF8String '" + hextoutf8(ASN1HEX.getHexOfV_AtObj(hex, idx)) + "'\n";
619     }
620     if (hex.substr(idx, 2) == "13") {
621 	return indent + "PrintableString '" + hextoutf8(ASN1HEX.getHexOfV_AtObj(hex, idx)) + "'\n";
622     }
623     if (hex.substr(idx, 2) == "14") {
624 	return indent + "TeletexString '" + hextoutf8(ASN1HEX.getHexOfV_AtObj(hex, idx)) + "'\n";
625     }
626     if (hex.substr(idx, 2) == "16") {
627 	return indent + "IA5String '" + hextoutf8(ASN1HEX.getHexOfV_AtObj(hex, idx)) + "'\n";
628     }
629     if (hex.substr(idx, 2) == "17") {
630 	return indent + "UTCTime " + hextoutf8(ASN1HEX.getHexOfV_AtObj(hex, idx)) + "\n";
631     }
632     if (hex.substr(idx, 2) == "18") {
633 	return indent + "GeneralizedTime " + hextoutf8(ASN1HEX.getHexOfV_AtObj(hex, idx)) + "\n";
634     }
635     if (hex.substr(idx, 2) == "30") {
636 	if (hex.substr(idx, 4) == "3000") {
637 	    return indent + "SEQUENCE {}\n";
638 	}
639 
640 	var s = indent + "SEQUENCE\n";
641 	var aIdx = ASN1HEX.getPosArrayOfChildren_AtObj(hex, idx);
642 
643 	var flagsTemp = flags;
644 	
645 	if ((aIdx.length == 2 || aIdx.length == 3) &&
646 	    hex.substr(aIdx[0], 2) == "06" &&
647 	    hex.substr(aIdx[aIdx.length - 1], 2) == "04") { // supposed X.509v3 extension
648 	    var oidHex = ASN1HEX.getHexOfV_AtObj(hex, aIdx[0]);
649 	    var oidDot = KJUR.asn1.ASN1Util.oidHexToInt(oidHex);
650 	    var oidName = KJUR.asn1.x509.OID.oid2name(oidDot);
651 
652 	    var flagsClone = JSON.parse(JSON.stringify(flags));
653 	    flagsClone.x509ExtName = oidName;
654 	    flagsTemp = flagsClone;
655 	}
656 	
657 	for (var i = 0; i < aIdx.length; i++) {
658 	    s = s + ASN1HEX.dump(hex, flagsTemp, aIdx[i], indent + "  ");
659 	}
660 	return s;
661     }
662     if (hex.substr(idx, 2) == "31") {
663 	var s = indent + "SET\n";
664 	var aIdx = ASN1HEX.getPosArrayOfChildren_AtObj(hex, idx);
665 	for (var i = 0; i < aIdx.length; i++) {
666 	    s = s + ASN1HEX.dump(hex, flags, aIdx[i], indent + "  ");
667 	}
668 	return s;
669     }
670     var tag = parseInt(hex.substr(idx, 2), 16);
671     if ((tag & 128) != 0) { // context specific 
672 	var tagNumber = tag & 31;
673 	if ((tag & 32) != 0) { // structured tag
674 	    var s = indent + "[" + tagNumber + "]\n";
675 	    var aIdx = ASN1HEX.getPosArrayOfChildren_AtObj(hex, idx);
676 	    for (var i = 0; i < aIdx.length; i++) {
677 		s = s + ASN1HEX.dump(hex, flags, aIdx[i], indent + "  ");
678 	    }
679 	    return s;
680 	} else { // primitive tag
681 	    var v = ASN1HEX.getHexOfV_AtObj(hex, idx);
682 	    if (v.substr(0, 8) == "68747470") { // http
683 		v = hextoutf8(v);
684 	    }
685 	    if (flags.x509ExtName === "subjectAltName" &&
686 		tagNumber == 2) {
687 		v = hextoutf8(v);
688 	    }
689 	    
690 	    var s = indent + "[" + tagNumber + "] " + v + "\n";
691 	    return s;
692 	}
693     }
694     return indent + "UNKNOWN(" + hex.substr(idx, 2) + ") " + 
695 	   ASN1HEX.getHexOfV_AtObj(hex, idx) + "\n";
696 };

698 /**
699  * check wheather the string is ASN.1 hexadecimal string or not
700  * @name isASN1HEX
701  * @memberOf ASN1HEX
702  * @function
703  * @param {String} hex string to check whether it is hexadecmal string for ASN.1 DER or not
704  * @return {Boolean} true if it is hexadecimal string of ASN.1 data otherwise false
705  * @since jsrsasign 4.8.3 asn1hex 1.1.6
706  * @description
707  * This method checks wheather the argument 'hex' is a hexadecimal string of
708  * ASN.1 data or not.
709  * @example
710  * ASN1HEX.isASN1HEX('0203012345') → true // PROPER ASN.1 INTEGER
711  * ASN1HEX.isASN1HEX('0203012345ff') → false // TOO LONG VALUE
712  * ASN1HEX.isASN1HEX('02030123') → false // TOO SHORT VALUE
713  * ASN1HEX.isASN1HEX('fa3bcd') → false // WRONG FOR ASN.1
714  */
715 ASN1HEX.isASN1HEX = function(hex) {
716     if (hex.length % 2 == 1) return false;
717 
718     var intL = ASN1HEX.getIntOfL_AtObj(hex, 0);
719     var tV = hex.substr(0, 2);
720     var lV = ASN1HEX.getHexOfL_AtObj(hex, 0);
721     var hVLength = hex.length - tV.length - lV.length;
722     if (hVLength == intL * 2) return true;
723 
724     return false;
725 };

727 /**
728  * get hexacedimal string from PEM format data<br/>
729  * @name pemToHex
730  * @memberOf ASN1HEX
731  * @function
732  * @param {String} s PEM formatted string
733  * @param {String} sHead PEM header string without BEGIN/END(OPTION)
734  * @return {String} hexadecimal string data of PEM contents
735  * @since jsrsasign 7.0.1 asn1hex 1.1.9
736  * @description
737  * This static method gets a hexacedimal string of contents 
738  * from PEM format data. You can explicitly specify PEM header 
739  * by sHead argument. 
740  * Any space characters such as white space or new line
741  * will be omitted.<br/>
742  * NOTE: Now {@link KEYUTIL.getHexFromPEM} and {@link X509.pemToHex}
743  * have been deprecated since jsrsasign 7.0.1. 
744  * Please use this method instead.
745  * @example
746  * ASN1HEX.pemToHex("-----BEGIN PUBLIC KEY...") → "3082..."
747  * ASN1HEX.pemToHex("-----BEGIN CERTIFICATE...", "CERTIFICATE") → "3082..."
748  * ASN1HEX.pemToHex(" \r\n-----BEGIN DSA PRIVATE KEY...") → "3082..."
749  */
750 ASN1HEX.pemToHex = function(s, sHead) {
751     if (s.indexOf("-----BEGIN ") == -1)
752         throw "can't find PEM header: " + sHead;
753 
754     if (sHead !== undefined) {
755         s = s.replace("-----BEGIN " + sHead + "-----", "");
756         s = s.replace("-----END " + sHead + "-----", "");
757     } else {
758         s = s.replace(/-----BEGIN [^-]+-----/, '');
759         s = s.replace(/-----END [^-]+-----/, '');
760     }
761     var sB64 = s.replace(/\s+/g, '');
762     var dataHex = b64tohex(sB64);
763     return dataHex;
764 };
